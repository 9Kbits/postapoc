is_murder_quest_target_of_prev_trigger = {
	PREVPREV = {
		has_quest = quest_the_men_in_black_assassination
		quest_target = { character = PREVPREV }
	}
}

is_murder_quest_target_of_root_trigger = {
	ROOT = {
		has_quest = quest_the_men_in_black_assassination
		quest_target = { character = PREVPREV }
	}
}

assassins_appropriate_target_trigger = {
	is_within_diplo_range = ROOT
	is_adult = yes
	is_quest_target = no

	OR = {
		is_ruler = no
		AND = {
			ROOT = {
				society_rank = {
					rank == 3
					society = the_men_in_black
				}
			}
			primary_title = { higher_real_tier_than = COUNT }
		}
		AND = {
			ROOT = {
				society_rank = {
					rank == 2
					society = the_men_in_black
				}
			}
			primary_title = { lower_real_tier_than = EMPEROR }
		}
		AND = {
			ROOT = {
				society_rank = {
					rank == 1
					society = the_men_in_black
				}
			}
			primary_title = { lower_real_tier_than = DUKE }
		}
	}
}

can_pay_mib_debt = {
	OR = {
		AND = {
			is_variable_equal = { which = mib_debt value = 50 }
			wealth = 50
		}
		AND = {
			is_variable_equal = { which = mib_debt value = 250 }
			wealth = 250
		}
		AND = {
			is_variable_equal = { which = mib_debt value = 300 }
			wealth = 300
		}
		AND = {
			is_variable_equal = { which = mib_debt value = 500 }
			wealth = 500
		}
	}
}

has_assassins_prerequisites = {
	age = 16
	conditional_tooltip = { #recruitment decision
		trigger = { FROM = { society_member_of = the_men_in_black } }
		OR = {
			FROM = { ai = no }
			hidden_trigger = { ai = no }
			religion_group = oldworld
		}
	}
	conditional_tooltip = { #showing interest via interface
		trigger = { NOT = { FROM = { society_member_of = the_men_in_black } } }
		OR = {
			true_religion = americanist
			religion_group = oldworld
			AND = {
				ai = no
				NOT = { trait = zealous }
				any_playable_ruler = {
					religion = americanist
					NOT = {
						distance_from_realm = { value = 300 who = ROOT }
					}
				}
			}
		}
	}
	OR = {
		is_female = no
		has_game_rule = {
			name = gender
			value = all
		}
	}
}
